
## Step 0　初期設定

```bash
# -*- coding: utf-8 -*-

import arcpy
import os
import re
from arcpy.sa import ExtractByMask, ExtractBand, ExtractMultiValuesToPoints

arcpy.CheckOutExtension("Spatial")
arcpy.env.overwriteOutput = True

# 基本設定

map_name = "Site7_9"

composite_raster_name = "Site7_9_CompositeBands"

# Band 1 = B, Band 2 = G, Band 3 = R, Band 4 = REDEDGE, Band 5 = NIR
band_settings = [
    (1, "B"),
    (2, "G"),
    (3, "R"),
    (4, "REDEDGE"),
    (5, "NIR"),
]

output_folder = r"C:\GIS_output\Site7_92_extract"
output_gdb_name = "Site7_92_extract.gdb"

output_gdb = os.path.join(output_folder, output_gdb_name)

os.makedirs(output_folder, exist_ok=True)

if not arcpy.Exists(output_gdb):
    arcpy.management.CreateFileGDB(output_folder, output_gdb_name)

arcpy.env.workspace = output_gdb

print("出力GDB:")
print(output_gdb)
```

## Step 1  マップ・ラスタ・ポリゴン確認

```bash
# 関数

def safe_name(name, max_len=80):
    name = os.path.splitext(os.path.basename(str(name)))[0]
    name = re.sub(r"[^\w\u3040-\u30ff\u3400-\u9fff]", "_", name)
    name = re.sub(r"_+", "_", name)
    name = name.strip("_")

    if not name:
        name = "unnamed"

    if re.match(r"^\d", name):
        name = "n_" + name

    return name[:max_len]


def unique_path(workspace, base_name):
    base_name = safe_name(base_name)
    out_path = os.path.join(workspace, base_name)

    if not arcpy.Exists(out_path):
        return out_path

    i = 1
    while True:
        candidate = os.path.join(workspace, f"{base_name}_{i}")
        if not arcpy.Exists(candidate):
            return candidate
        i += 1


# マップ取得

aprx = arcpy.mp.ArcGISProject("CURRENT")

maps = aprx.listMaps(map_name)

if not maps:
    print("現在のプロジェクト内のマップ一覧:")
    for mm in aprx.listMaps():
        print(f"  - {mm.name}")
    raise ValueError(f"マップ '{map_name}' が見つかりません。")

m = maps[0]

print(f"対象マップ: {m.name}")


# ラスタ取得

composite_raster_path = None

print("\n=== ラスタレイヤー一覧 ===")

for lyr in m.listLayers():
    if lyr.isRasterLayer:
        desc = arcpy.Describe(lyr)
        print("-----------------------------")
        print("Layer name :", lyr.name)
        print("CatalogPath:", desc.catalogPath)

        if lyr.name == composite_raster_name:
            composite_raster_path = desc.catalogPath

if composite_raster_path is None:
    raise ValueError(f"元ラスタ '{composite_raster_name}' が見つかりません。")

print("\n元マルチバンドラスタ:")
print(composite_raster_path)


# ポリゴンレイヤー取得

polygon_layers = []

print("\n=== ポリゴンレイヤー一覧 ===")

for lyr in m.listLayers():
    if lyr.isFeatureLayer:
        desc = arcpy.Describe(lyr)
        if hasattr(desc, "shapeType") and desc.shapeType == "Polygon":
            polygon_layers.append(lyr)
            print(f"  - {lyr.name}")

if not polygon_layers:
    raise ValueError("ポリゴンレイヤーが見つかりません。")

print(f"\nポリゴンレイヤー数: {len(polygon_layers)}")
```

## Step 2  ポリゴンを1個ずつマスク用FCとして保存

```bash
# Step 2. 各ポリゴンを個別マスクとして保存

mask_fcs = []

for poly_lyr in polygon_layers:

    poly_desc = arcpy.Describe(poly_lyr)
    poly_path = poly_desc.catalogPath
    oid_field = poly_desc.OIDFieldName
    poly_layer_name = safe_name(poly_lyr.name)

    print(f"\n処理中のポリゴンレイヤー: {poly_lyr.name}")

    with arcpy.da.SearchCursor(poly_lyr, [oid_field]) as cursor:
        for row in cursor:
            oid = row[0]

            mask_name = f"mask_{poly_layer_name}_{oid}"
            mask_fc = os.path.join(output_gdb, safe_name(mask_name))

            if arcpy.Exists(mask_fc):
                print(f"  既存のためスキップ: {os.path.basename(mask_fc)}")
                mask_fcs.append(mask_fc)
                continue

            where = f"{arcpy.AddFieldDelimiters(poly_path, oid_field)} = {oid}"

            temp_layer = arcpy.management.MakeFeatureLayer(
                poly_lyr,
                f"temp_{mask_name}",
                where
            )[0]

            arcpy.management.CopyFeatures(temp_layer, mask_fc)

            mask_fcs.append(mask_fc)

            print(f"  作成: {os.path.basename(mask_fc)}")

print("\nマスク作成完了")
print(f"マスク数: {len(mask_fcs)}")
```

## Step 3  マスクごとにComposite rasterを切り抜き

```bash
# Step 3. Composite rasterをマスクで切り抜き

clipped_composites = []

for mask_fc in mask_fcs:

    mask_base = safe_name(os.path.basename(mask_fc))
    out_name = f"{mask_base}_Composite_clip"
    out_raster = os.path.join(output_gdb, safe_name(out_name))

    if arcpy.Exists(out_raster):
        print(f"既存のためスキップ: {os.path.basename(out_raster)}")
        clipped_composites.append(out_raster)
        continue

    print(f"\nExtractByMask 開始: {mask_base}")

    clipped = ExtractByMask(composite_raster_path, mask_fc)
    clipped.save(out_raster)

    clipped_composites.append(out_raster)

    print(f"完了: {os.path.basename(out_raster)}")

print("\nComposite raster の切り抜き完了")
print(f"出力数: {len(clipped_composites)}")
```

## Step 4  切り抜きラスタからB/G/R/REDEDGE/NIRを抽出

```bash
# Step 4. 各Composite clipからB/G/R/REDEDGE/NIRを抽出

band_raster_dict = {}

for clipped_composite in clipped_composites:

    comp_base = safe_name(os.path.basename(clipped_composite))
    mask_base = comp_base.replace("_Composite_clip", "")

    band_raster_dict[mask_base] = {}

    print(f"\nバンド抽出開始: {mask_base}")

    for band_number, band_name in band_settings:

        out_name = f"{mask_base}_{band_name}_clip"
        out_band = os.path.join(output_gdb, safe_name(out_name))

        if arcpy.Exists(out_band):
            print(f"  既存のためスキップ: {os.path.basename(out_band)}")
            band_raster_dict[mask_base][band_name] = out_band
            continue

        print(f"  ExtractBand: Band {band_number} -> {band_name}")

        band_raster = ExtractBand(
            in_raster=clipped_composite,
            band_ids=[band_number]
        )

        band_raster.save(out_band)

        band_raster_dict[mask_base][band_name] = out_band

        print(f"  完了: {os.path.basename(out_band)}")

print("\nバンド抽出完了")
```

## Step 5  1つのバンドをRaster to Point

```bash
# Step 5. BバンドをRaster to Point

point_fcs = []

for mask_base, bands in band_raster_dict.items():

    b_raster = bands["B"]

    out_name = f"{mask_base}_points"
    out_points = os.path.join(output_gdb, safe_name(out_name))

    if arcpy.Exists(out_points):
        print(f"既存のためスキップ: {os.path.basename(out_points)}")
        point_fcs.append(out_points)
        continue

    print(f"\nRasterToPoint 開始: {mask_base}")

    arcpy.conversion.RasterToPoint(
        in_raster=b_raster,
        out_point_features=out_points,
        raster_field="VALUE"
    )

    point_fcs.append(out_points)

    print(f"完了: {os.path.basename(out_points)}")

print("\nRaster to Point 完了")
print(f"ポイントFC数: {len(point_fcs)}")
```

## Step 6  ポイントにB/G/R/REDEDGE/NIRの値を追加

```bash
# Step 6. Extract Multi Values to Points

final_point_outputs = []

for point_fc in point_fcs:

    point_base = safe_name(os.path.basename(point_fc))
    mask_base = point_base.replace("_points", "")

    if mask_base not in band_raster_dict:
        print(f"対応するバンドラスタが見つからないためスキップ: {point_base}")
        continue

    bands = band_raster_dict[mask_base]

    print(f"\nExtractMultiValuesToPoints 開始: {point_base}")

    raster_value_list = [
        [bands["B"], "B"],
        [bands["G"], "G"],
        [bands["R"], "R"],
        [bands["REDEDGE"], "REDEDGE"],
        [bands["NIR"], "NIR"],
    ]

    ExtractMultiValuesToPoints(
        in_point_features=point_fc,
        in_rasters=raster_value_list,
        bilinear_interpolate_values="NONE"
    )

    final_point_outputs.append(point_fc)

    print(f"完了: {os.path.basename(point_fc)}")

print("\nすべて完了")
print("最終ポイント:")

for fc in final_point_outputs:
    print(fc)
```
